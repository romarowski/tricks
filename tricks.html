VIM browse files in visual mode (equivalent to l) :e [/directory/]

'>' line only in file2
'<' line only on file1
python datetime get current time in good format now = datetime.datetime.now()

pip install local package with symlink pip install -e [/dir2pkg/]
installing *.tar.gz install compressed binaries in linux tar -xvzf tool_v0.1.tar.gz
./tool_v0.1.tar.gz #typically is an executable

after installation move the file to /opt/ and create symlink to /usr/bin

sudo mv tool_v0.1.tar.gz /opt/
sudo ln -s /opt/tool_v0.1.tar.gz /usr/bin/tool
